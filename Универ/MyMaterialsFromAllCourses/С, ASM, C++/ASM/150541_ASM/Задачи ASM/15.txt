;ѕример 15
;ѕодсчитать количество четных элементов массива чисел и найти их сумму. 
;ћассив задан в сегменте данных 

.model small
.stack	100h
.data

len equ	15	;количество элементов в mas
mas	db	1,6,3,9,2,8,9,6,12,7,8,15,7,2,5
summa	db	?	;здесь будет сумма четных чисел
kol	db	?	;здесь будет количество четных чисел

.code
start:
	mov	ax,@data
	mov	ds,ax
	mov	cx,len		;длина mas в cx
	xor	ax,ax	
	xor 	si,si	
	xor	bx,bx
	xor	dx,dx
	mov	bl,2

m1:	mov	al,mas[si] 	;в al помещаем очередной элемент массива
	mov	bh,al		;сохран€ем его в bh
	div	bl		;делим его на два
	cmp	ah,0		;остаток равен нулю?(элемент четный)
	jne	m2		;нет - продолжаем, переход на m2
	add	dl,bh 		;да - добавл€ем очередной четный элемент в dl 
	inc	dh		;и подсчитываем количество четных чисел
m2:
	inc 	si		;переходим к следующему элементу массива
	loop	m1		;замыкаем цикл

	mov	summa,dl	;сумма из dl записываем в пам€ть
	mov	kol,dh		;количество чисел записываем в пам€ть
exit:
	mov	ax,4c00h
	int	21h	;возврат управлени€ операционной системе
end	start
