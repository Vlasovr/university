;ѕример 12
;Ёлементы массива символов, €вл€ющиес€ строчными буквами латинского алфавита,
;заменить на одноименные прописные

.model	small
.stack	100h
.data
n	equ	12
stroka	db	'acC9df0gnNdu'

.code
start:
	mov	ax,@data
	mov	ds,ax
	xor	ax,ax
	xor     dx,dx   ;в dx будет сумма цифр
	xor     bx,bx   ;в bx - их количество
	mov	cx,n
	lea	bx,stroka	;адрес stroka в bx
m1:	mov	al,[bx]	;очередной символ из stroka в al
	cmp	al,'a'	;проверить, что код символа не меньше 'a'
	jb	next	;если меньше, то на следующий символ
	cmp	al,'z'	;проверить, что код символа не больше 'z'
	ja	next	;если больше, то на следующий символ

;преобразуем прописную букву в строчную

	add     al,'A'-'a'
	mov	[bx],al  ;записываем измененный символ на прежнее место

next:
	inc	bx	;адресовать следующий символ
	dec	cx	;уменьшить значение счетчика в cx
	jnz	m1	;если cx не 0, то переход на m1
exit:
	mov	ax,4c00h
	int	21h	;возврат управлени€ операционной системе
end	start
