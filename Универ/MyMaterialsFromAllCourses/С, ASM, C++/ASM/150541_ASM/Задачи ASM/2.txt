
;Пример2
;Найти среднее арифметическое элементов массива чисел

.model small
.stack	100h
.data
len equ	10	;количество элементов в mas
mas	db	1,0,9,8,0,7,8,0,2,0
.code
start:
	mov	ax,@data
	mov	ds,ax
	mov	cx,len	;длину поля mas в cx
	xor	ax,ax
	xor 	si,si
	jcxz	exit	;проверка cx на 0, если 0, то на выход
cycl:
	add 	al,mas[si]   ;находим сумму элементов массива
	inc	si	   ;перейти к следующему элементу
	loop	cycl

	mov 	bl,len
	div	bl

; в al находится среднее арифметическое элементов массива

exit:
	mov	ax,4c00h
	int	21h	;возврат управления операционной системе

end	start
