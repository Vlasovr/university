;Пример 5
;Найти суммы положительных и отрицательных элементов массива,определенных в сегменте кода 

.model tiny
.code
org 100h
begin:
	jmp continue

len equ	20	;количество элементов в mas

;массив чисел

massiv	db	1,-3,-5,0,-29,8,4,-7,0,7,-5,-3,8,7,-4,0,2,-5,0,1

pozitiv	dw	?	;место для записи суммы положительных чисел
negativ	dw	?	;место для записи суммы отрицательных чисел

continue:

 xor 	bx,bx	;для адресации элементов массива
 xor	dx,dx	;здесь будет сумма положительных 			;чисел
	xor	bp,bp	;здесь будет сумма 			;отрицательных чисел
	mov	cx,20	;счетчик циклов
	xor	ax,ax

l1:
	mov	al,mas[bx];добавляем очередной элемент массива к сумме
	cmp	al,0
	jae	poz
	add	bp,ax
	jmp	next
poz:	
	add	dx,ax

next:	
	inc	bx	;переходим к следующему 			;элементу массива
	loop 	l1

	mov	pozitiv,dx ;запись положительной 			;суммы в память
	mov	negativ,bp  ;запись отрицательной 				;суммы в память

exit:		;выход
	mov	ax,4c00h
	int	21h
end	begin
