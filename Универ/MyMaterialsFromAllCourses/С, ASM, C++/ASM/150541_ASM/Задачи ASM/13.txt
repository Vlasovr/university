;¬ массиве чисел, определенном в сегменте данных,отрицательные значени€ 
;заменить нул€ми, а положительные просуммировать

.model small
.stack	100h
.data

len equ	15	;количество элементов в mas
mas	db	1,-3,3,9,-5,8,9,-2,-3,7,8,-8,1,2,5
summa	db	?	;здесь будет сумма положительных чисел


.code
start:
	mov	ax,@data
	mov	ds,ax
	mov	cx,len		;длина mas в cx
	xor	ax,ax	
	xor 	si,si	
	xor	bx,bx

m1:	mov	al,mas[si] 	;в al помещаем очередной элемент массива
	cmp	al,0		;al > 0
	ja	m2		;да - переход на m2
	mov	mas[si],0	;нет - на его место записываем ноль
m2:
	add	dl,al		;добавл€ем очередной положительный элемент в dl 
	inc 	si		;переходим к следующему элементу массива
	loop	m1		;замыкаем цикл

	mov	summa,dl	;сумма из dl записываем в пам€ть

exit:
	mov	ax,4c00h
	int	21h	;возврат управлени€ операционной системе
end	start
