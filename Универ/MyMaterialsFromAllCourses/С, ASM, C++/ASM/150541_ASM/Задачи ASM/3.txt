;Пример 3
;Вычислить сумму элементов массива чисел, определенных в сегменте кода 

.model tiny
.code
org 100h
begin:
	jmp continue

len equ	10	;количество элементов в mas
mas	db	1,0,9,8,0,7,8,0,2,0;тексты сообщений
summa 	dw	? ;место для записи полученной суммы

continue:

	xor 	bx,bx	;для адресации элементов 
				;массива
	xor	dx,dx	;здесь будет сумма чисел
	mov	cx,10	;счетчик циклов


l1:
	add	dl,mas[bx];добавляем очередной                           ;элемент массива к сумме
	inc	bx	;переходим к следующему                         ;элементу массива
	loop 	l1	;

	mov	summa,dx;запись суммы в память

exit:		;выход
	mov	ax,4c00h
	int	21h
end	begin
