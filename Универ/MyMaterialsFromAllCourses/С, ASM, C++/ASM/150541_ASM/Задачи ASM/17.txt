;Пример 17
;В массиве чисел, определенном в сегменте данных,подсчитать количество отрицательных,
;положительных и нулевых элементов

.model small
.stack	100h
.data

len equ	15	;количество элементов в mas
mas	db	1,-3,0,9,-5,8,9,-2,-3,7,8,-8,0,2,5
pozitiv	db	?	;здесь будет количество положительных чисел
negativ db	?	;здесь будет количество отрицательных чисел
nuls	db	?	;здесь будет количество нулевых чисел

.code
start:
	mov	ax,@data
	mov	ds,ax
	mov	cx,len		;длина mas в cx
	xor	ax,ax	
	xor 	si,si	
	xor	bx,bx

m1:	mov	al,mas[si] 	;в al помещаем очередной элемент массива
	cmp	al,0		;al = 0?
	je	m3		;да - переход на m3
	ja	m2		;al>0 - переход на m2
	inc	bl		;иначе в bl подсчитываем отрицательные элементы
	jmp	bb		;продолжаем
m2:
	inc	bh		;в bh подсчитываем положительные элементы
	jmp	bb		;продолжаем
m3:
	inc	ah		;в ah подсчитываем нули
bb: 
	inc 	si		;переходим к следующему элементу массива
	loop	m1		;замыкаем цикл

	mov	nuls,ah		;содержимое ah(количество нулей) записываем в память
	mov	pozitiv,bh	;содержимое bh(количество положительных чисел) записываем в память
	mov	ntgativ,bl	;содержимое bl(количество отрицательных чисел) записываем в память
exit:
	mov	ax,4c00h
	int	21h	;возврат управления операционной системе
end	start
