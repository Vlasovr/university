Необходимо проверить, является ли слово палиндромом (т.е. читается
одинаково с обеих концов). Например ковок(kovok), нилин(nilin),
потоп(potop)


//Функция проверка слова на палиндромность
int palindrom(char *str) 
{ 
   int  left_index = 0;
   int right_index = strlen(str)-1;
   //0 - истина, 1 - ложь
   while(left_index<right_index) 
     if(str[left_index++]!=str[right_index--]) return 1;
     return 0;
};
