#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#define N    4    
#define N1    N+1 
 
float matrix[N][N1]={{10.800,-0.576,0.000,0.000,12.143},
                     {0.321,9.900,7.300,0.000,13.086},
                     {0.000,0.369,9.000,-6.060,13.674},
                     {0.000,0.000,0.416,8.100,13.897}
                    };
 
float epsilon=0.001;
 
void ShowMatrix(void)
{ printf("SLAR: \n");
     int i,j;
  for (i=0;i<N;i++)
  {
    for (j=0;j<N;j++)
      printf("%+3.3f*x%d",matrix[i][j],i+1);
    printf("=%3.3f\n",matrix[i][N]);
  }
}
 
int main()
{
  float tmp,xx[N1];
  short int i,j,k;
 
  ShowMatrix();
  
  /*Метод Гаусса*/
  
 for (i=0;i<N;i++)
  {
    tmp=matrix[i][i];
    for (j=N;j>=i;j--) matrix[i][j]/=tmp;
    for (j=i+1;j<N;j++)
    {
      tmp=matrix[j][i];
      for (k=N;k>=i;k--)
        matrix[j][k]-=tmp*matrix[i][k];
    }
  }
  xx[N-1]=matrix[N-1][N];
  for (i=N-2;i>=0;i--)
  {
    xx[i]=matrix[i][N];
    for (j=i+1;j<N;j++) xx[i]-=matrix[i][j]*xx[j];
  }
 
printf("\nMetod Gaussa:\n");
  for (i=0;i<N;i++)
    printf("x%d=%3.3f\n",i+1,xx[i]);
 
 system("pause");
 return 0;
}
