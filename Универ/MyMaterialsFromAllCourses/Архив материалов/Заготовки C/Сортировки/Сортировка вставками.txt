int i, j, temp;
// цикл проходов, i - номер прохода
for (i = 1; i < size; i++) 
{
  // поиск места элемента в готовой последовательности 
  temp = array[i];
  // сдвигаем элемент направо, пока не дошли
  for (j = i - 1; j >= 0; j--) 
  {
    if (array[j] < temp) {break;}
    array[j+1] = array[j];
  }
  // место найдено, вставить элемент
  array[j+1] = temp;
}
